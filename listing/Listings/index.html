    struct Listing {
  struct Assets.Asset asset;
  struct Listings.Params params;
  address lister;
  uint32 maxLockPeriod;

}

    Registry

    struct Registry {
  struct CountersUpgradeable.Counter listingIdTracker;
  contract IListingStrategyRegistry strategyRegistry;
  struct EnumerableSetUpgradeable.UintSet listingIndex;
  mapping(uint256 &#x3D;&gt; struct Listings.Listing) listings;
  mapping(address &#x3D;&gt; struct Listings.ListerInfo) listers;
  mapping(address &#x3D;&gt; struct Listings.AssetInfo) assets;
  struct CountersUpgradeable.Counter listingGroupIdTracker;
  mapping(uint256 &#x3D;&gt; struct Listings.ListingGroupInfo) listingGroups;
}
